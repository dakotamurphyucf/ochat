Module Io.Task_pool

A minimal domain-aware worker pool.

Usage

  module Pool = Io.Task_pool (struct
    type input  = int
    type output = int
    let dm     = domain_mgr
    let stream = Eio.Stream.create 0
    let sw     = switch
    let handler x = x * x
  end)

  let () = Pool.spawn "square" in
  assert (Pool.submit 11 = 121)
