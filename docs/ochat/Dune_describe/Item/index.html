type module_ = private {
  1. name : string;
  2. impl : string option;
  3. intf : string option;
  4. cmt : string option;
  5. cmti : string option;
}
type library = private {
  1. name : string;
  2. uid : string;
  3. requires : string list;
  4. local : bool;
  5. source_dir : string;
  6. modules : module_ list;
  7. include_dirs : string list;
}
type executable = private {
  1. names : string list;
  2. requires : string list;
  3. modules : module_ list;
  4. include_dirs : string list;
}
