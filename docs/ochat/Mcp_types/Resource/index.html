Module Mcp_types.Resource

Metadata for a single resource returned by "resources/list".

Only the fields actively required by the client are represented. The server is free to attach additional keys; they will be ignored during deserialisation.

type t = {
  1. uri : string;
    (*

    Globally unique resource URI.

    *)
  2. name : string;
    (*

    Display name.

    *)
  3. description : string option;
    (*

    Optional free-form summary.

    *)
  4. mime_type : string option;
    (*

    Media type (e.g. "text/plain", "image/png").

    *)
  5. size : int option;
    (*

    Size in bytes, when known.

    *)
}
include Ppx_jsonaf_conv_lib.Jsonafable.S with type t := t
val t_of_jsonaf : Jsonaf_kernel__.Type.t -> t
val jsonaf_of_t : t -> Jsonaf.t
