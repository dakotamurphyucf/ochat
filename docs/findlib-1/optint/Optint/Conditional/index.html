Module Optint.Conditional

A conditional type equality, used for revealing that a type t has one of two possible implementation types u and v.

type ('t, 'u, 'v) t =
  1. | True : ('t, 't, _) t
    (*

    therefore 't = 'u

    *)
  2. | False : ('t, _, 't) t
    (*

    therefore 't = 'v

    *)
